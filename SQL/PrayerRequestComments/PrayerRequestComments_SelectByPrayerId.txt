ALTER PROC [dbo].[PrayerRequestComments_SelectByPrayerId]
        @PrayerId INT
AS
/* -- TESTING EXECUTION BLOCK
        DECLARE @PrayerId INT = 1;
        EXEC dbo.PrayerRequestComments_SelectByPrayerId @PrayerId;
*/
BEGIN
        SELECT
                CommentId,
                PrayerId,
                UserId,
                Comment,
                DateCreated
        FROM dbo.PrayerRequestComments
        WHERE PrayerId = @PrayerId
        ORDER BY DateCreated ASC;
END;
