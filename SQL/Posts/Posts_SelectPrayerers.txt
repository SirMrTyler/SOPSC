ALTER PROC [dbo].[Posts_SelectPrayerers]
    @PrayerId INT
AS
/* -- TESTING EXECUTION BLOCK
    DECLARE @PrayerId INT = 1;

    EXEC dbo.Posts_SelectPrayerers @PrayerId;
*/
BEGIN
    SELECT
        pp.UserId,
        u.FirstName + ' ' + u.LastName AS [Name]
    FROM dbo.PostPrayers pp
    JOIN dbo.Users u ON pp.UserId = u.UserId
    WHERE pp.PrayerId = @PrayerId;
END;
